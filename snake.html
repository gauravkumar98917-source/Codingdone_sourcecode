Snake Game Code

from tkinter import *
from tkinter import messagebox
import random
import json

# ----- Constants and initial setup -----
Cell = 20
snake = [(100, 100), (80, 100), (60,100)]
direction = "right"
food = (160, 200)
score = 0
try:
    with open("highscore.json", "r") as file:
        high_score = json.load(file)
except:
    high_score = 0
after_id = None

# ----- Direction functions -----
def up():
    global direction
    if direction != "down":
        direction = "up"

def down():
    global direction
    if direction != "up":
        direction = "down"

def left():
    global direction
    if direction != "right":
        direction = "left"

def right():
    global direction
    if direction != "left":
        direction = "right"

# ----- Snake movement function -----
def snake_move():
    global score, food, high_score, after_id
    head_x, head_y = snake[0]

    # Calculate new head based on direction
    if direction == "right":
        new_head = (head_x + Cell, head_y)
    elif direction == "left":
        new_head = (head_x - Cell, head_y)
    elif direction == "up":
        new_head = (head_x, head_y - Cell)
    elif direction == "down":
        new_head = (head_x, head_y + Cell)

    # Wrap around screen
    new_head = (new_head[0] % 700, new_head[1] % 1200)

    # Check self-collision
    if new_head in snake[1:]:
        answer = messagebox.askyesno("Game Over", "GAME OVER\nDo you want to restart?")
        if answer:
            restart_game()
        return

    # Insert new head
    snake.insert(0, new_head)

    # Check food collision
    if new_head == food:
        score += 1
        x = random.randint(0, 29) * Cell
        y = random.randint(0, 59) * Cell
        food = (x, y)
        if score > high_score:
            high_score = score
            with open("highscore.json", "w")as file:
                json.dump(high_score, file)
    else:
        snake.pop()

    # Clear canvas
    canvas.delete("all")

    # ----- Draw grid -----
    for i in range(0, 700, Cell):
        canvas.create_line(i, 0, i, 1200, fill="grey20")
    for j in range(0, 1200, Cell):
        canvas.create_line(0, j, 700, j, fill="gray20")

    # ----- Draw score -----
    canvas.create_text(50, 10, text=f"Score: {score}", fill="white", font=("Helvetica", 8), anchor="nw")
    canvas.create_text(400, 10, text=f"High Score: {high_score}", fill="cyan", font=("Arial", 8), anchor="nw")

    # ----- Draw snake -----
    for index, block in enumerate(snake):
        x, y = block
        if index == 0:
            canvas.create_rectangle(x, y, x + Cell, y + Cell, fill="limegreen", outline="darkgreen", width=2)
        else:
            canvas.create_rectangle(x, y, x + Cell, y + Cell, fill="lime", outline="darkgreen", width=2)

    # ----- Draw food -----
    fx, fy = food
    canvas.create_rectangle(fx, fy, fx + Cell, fy + Cell, fill="red", outline="maroon", width=2)

    # Repeat movement
    after_id = root.after(100, snake_move)

# ----- Restart function -----
def restart_game():
    global snake, direction, food, score, after_id
    if after_id is not None:
        root.after_cancel(after_id)
        after_id = None

    snake = [(100, 100), (80, 100), (60, 100)]
    direction = "right"
    score = 0
    food = (160, 200)

    # Clear canvas
    canvas.delete("all")

    # Draw initial score
    canvas.create_text(50, 20, text=f"Score: {score}", fill="white", font=("Arial", 16), anchor="nw")

    # Draw initial snake
    for block in snake:
        x, y = block
        canvas.create_rectangle(x, y, x + Cell, y + Cell, fill="lime", outline="darkgreen", width=2)

    # Draw initial food
    fx, fy = food
    canvas.create_rectangle(fx, fy, fx + Cell, fy + Cell, fill="red", outline="maroon", width=2)

    snake_move()

# ----- Main window -----
root = Tk()
root.geometry("700x1200")

# ----- Canvas -----
canvas = Canvas(root, width=700, height=1200, bg="black")
canvas.pack()

# ----- Controls -----
controls = Frame(root)
controls.pack()

btn_up = Button(controls, text="↑", command=up, width=3, height=2)
btn_down = Button(controls, text="↓", command=down, width=3, height=2)
btn_left = Button(controls, text="←", command=left, width=3, height=2)
btn_right = Button(controls, text="→", command=right, width=3, height=2)

btn_up.grid(row=0, column=1)
btn_left.grid(row=1, column=0)
btn_down.grid(row=1, column=1)
btn_right.grid(row=1, column=2)

# ----- Start game -----
restart_game()

# ----- Main loop -----
root.mainloop()")
