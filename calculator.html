from tkinter import*

def add(value):
    entry.insert("end", value)
    
def clear():
	entry.delete(0, END)
	
def backspace():
    current = entry.get()  
    entry.delete(0, END)   
    entry.insert(0, current[:-1]) 
	
def calculate():
	try:
		result=eval(entry.get())
		entry.delete(0, END)
		entry.insert("end", result)
	except ZeroDivisionError:
		entry.delete(0, END)
		entry.insert(0, "Arthemitcal error")
	except:
		entry.delete(0, END)
		entry.insert(0, "Syntax error")
