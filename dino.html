Dino Game - Python Pygame

import pygame
import random
pygame.init()

clock = pygame.time.Clock()

cloud_x = 500
cloud_y = 100
game_over = False

dino_x = 100
dino_y = 300
dino_width = 40
dino_height = 50

velocity = 0
gravity = 0.8
jump = -15
on_ground = True

cactus_x = 800
cactus_y = 310
cactus_width = 20
cactus_height = 40
cactus_speed = 5

score = 0
font = pygame.font.SysFont(None, 36)
big_font = pygame.font.SysFont(None, 72)

screen = pygame.display.set_mode((800, 400))
pygame.display.set_caption("Dino Game")

running = True

while running:

    # --------- GAME UPDATE ---------
    if not game_over:
        velocity += gravity
        dino_y += velocity

        if dino_y >= 300:
            dino_y = 300
            velocity = 0
            on_ground = True

        cactus_x -= cactus_speed
        if cactus_x < -20:
            cactus_x = random.randint(800, 1200)

        score += 0.1
        cactus_speed = 5 + int(score / 100)

        cloud_x -= 2
        if cloud_x < -100:
            cloud_x = 900

    # --------- COLLISION ---------
    dino_rect = pygame.Rect(dino_x, dino_y, dino_width, dino_height)
    cactus_rect = pygame.Rect(cactus_x, cactus_y, cactus_width, cactus_height)

    if dino_rect.colliderect(cactus_rect):
        game_over = True

    # --------- EVENTS ---------
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

        if event.type == pygame.MOUSEBUTTONDOWN:
            if game_over:
                dino_y = 300
                cactus_x = 800
                score = 0
                velocity = 0
                cactus_speed = 5
                game_over = False
            else:
                if on_ground:
                    velocity = jump
                    on_ground = False

    # --------- DRAWING ---------
    screen.fill((135, 206, 235))  # Sky

    # Cloud
    pygame.draw.circle(screen, (255, 255, 255), (int(cloud_x), cloud_y), 20)
    pygame.draw.circle(screen, (255, 255, 255), (int(cloud_x + 20), cloud_y + 5), 20)
    pygame.draw.circle(screen, (255, 255, 255), (int(cloud_x - 20), cloud_y + 5), 20)

    # ---- 2D Ground (your layers) ----
    pygame.draw.rect(screen, (0, 150, 0), (0, 350, 800, 50))    # Grass top layer
    pygame.draw.rect(screen, (139, 69, 19), (0, 400, 800, 400))   # Dirt middle layer
    pygame.draw.rect(screen, (128, 128, 128), (0, 800, 800, 800)) # Stone bottom layer

    # ---- Dino Drawing (Chrome Style) ----
    pygame.draw.rect(screen, (83, 83, 83), (dino_x, dino_y, 40, 30))          # Body
    pygame.draw.rect(screen, (83, 83, 83), (dino_x+30, dino_y-10, 20, 20))    # Head
    pygame.draw.rect(screen, (255, 255, 255), (dino_x+45, dino_y-5, 5, 5))    # Eye
    pygame.draw.rect(screen, (83, 83, 83), (dino_x-10, dino_y+10, 10, 5))     # Tail
    pygame.draw.rect(screen, (83, 83, 83), (dino_x+5, dino_y+30, 8, 15))      # Leg 1
    pygame.draw.rect(screen, (83, 83, 83), (dino_x+20, dino_y+30, 8, 15))     # Leg 2

    # ---- Cactus Drawing (Chrome Style) ----
    pygame.draw.rect(screen, (0, 153, 0), (cactus_x, cactus_y, 20, 40))      # Main body
    pygame.draw.rect(screen, (0, 153, 0), (cactus_x - 10, cactus_y + 10, 10, 10)) # Left arm
    pygame.draw.rect(screen, (0, 153, 0), (cactus_x + 20, cactus_y + 15, 10, 10)) # Right arm

    # Score
    score_text = font.render("Score: " + str(int(score)), True, (0, 0, 0))
    screen.blit(score_text, (550, 20))

    # Game Over
    if game_over:
        game_over_text = big_font.render("GAME OVER", True, (200, 0, 0))
        screen.blit(game_over_text, (220, 150))

    pygame.display.update()
    clock.tick(60)

pygame.quit()
